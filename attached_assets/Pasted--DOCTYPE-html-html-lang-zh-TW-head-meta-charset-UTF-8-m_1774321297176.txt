    <h2>➕ 新增消費</h2>
    <form id="billForm">
        <div class="form-group">
            <label>消費日期</label>
            <input type="date" id="date" required>
        </div>
        
        <div class="form-group">
            <label>消費項目</label>
            <input type="text" id="item" placeholder="例如：晚餐、衛生紙、電費" required>
        </div>
        
        <div class="form-group">
            <label>總金額 ($)</label>
            <input type="number" id="amount" placeholder="0" required>
        </div>
        
        <div class="form-group">
            <label>誰先付錢？（付款人）</label>
            <div class="radio-group">
                <label class="radio-option selected"><input type="radio" name="paidBy" value="我" checked onchange="updateRadioStyling()"> 我</label>
                <label class="radio-option"><input type="radio" name="paidBy" value="室友" onchange="updateRadioStyling()"> 室友</label>
            </div>
        </div>
        
        <div class="form-group">
            <label>這筆錢怎麼算？（分攤模式）</label>
            <div class="radio-group">
                <label class="radio-option selected"><input type="radio" name="splitMode" value="均分" checked onchange="updateRadioStyling()"> 大家均分</label>
                <label class="radio-option"><input type="radio" name="splitMode" value="只有我" onchange="updateRadioStyling()"> 只有我付</label>
                <label class="radio-option"><input type="radio" name="splitMode" value="只有室友" onchange="updateRadioStyling()"> 只有室友付</label>
            </div>
        </div>
        
        <button type="submit">記起來</button>
    </form>
    
    <div id="history">
        <h2>📜 歷史紀錄</h2>
        <div id="recordList"></div>
        <button onclick="clearAllData()" style="background-color: #636e72; margin-top: 20px; font-size: 14px;">清空所有資料</button>
    </div>

<script>
    // 設定預設日期為今天
    document.getElementById('date').valueAsDate = new Date();

    let records = JSON.parse(localStorage.getItem('roommate_bills')) || [];

    // 初始化介面
    updateUI();

    // 處理表單送出
    document.getElementById('billForm').addEventListener('submit', function(e) {
        e.preventDefault();
        
        const date = document.getElementById('date').value;
        const item = document.getElementById('item').value;
        const amount = parseFloat(document.getElementById('amount').value);
        const paidBy = document.querySelector('input[name="paidBy"]:checked').value;
        const splitMode = document.querySelector('input[name="splitMode"]:checked').value;

        if (!date || !item || isNaN(amount) || amount <= 0) {
            alert('請填寫完整的正確資訊');
            return;
        }

        const newRecord = {
            id: Date.now(),
            date: date,
            item: item,
            amount: amount,
            paidBy: paidBy,
            splitMode: splitMode
        };

        records.push(newRecord);
        saveData();
        updateUI();
        
        // 清空表單項目與金額
        document.getElementById('item').value = '';
        document.getElementById('amount').value = '';
        document.getElementById('date').valueAsDate = new Date();
    });

    // 計算結算金額與更新 UI
    function updateUI() {
        const listEl = document.getElementById('recordList');
        const balanceEl = document.getElementById('balance');
        listEl.innerHTML = '';
        
        let netBalance = 0; // 正數代表室友欠我，負數代表我欠室友

        // 按時間倒序顯示 (最新的在上面)
        const sortedRecords = [...records].sort((a, b) => b.id - a.id);

        sortedRecords.forEach(record => {
            // 計算這筆交易對結算的影響
            let share_me = 0;
            let share_roommate = 0;

            if (record.splitMode === '均分') {
                share_me = record.amount / 2;
                share_roommate = record.amount / 2;
            } else if (record.splitMode === '只有我') {
                share_me = record.amount;
                share_roommate = 0;
            } else if (record.splitMode === '只有室友') {
                share_me = 0;
                share_roommate = record.amount;
            }

            // 根據付款人計算債務變更
            if (record.paidBy === '我') {
                // 我付了全部，室友欠我他的分攤額
                netBalance += share_roommate;
            } else {
                // 室友付了全部，我欠室友我的分攤額
                netBalance -= share_me;
            }

            // 顯示紀錄清單
            const itemEl = document.createElement('div');
            itemEl.className = 'record-item';
            
            // 簡化顯示分攤模式
            let splitText = record.splitMode;
            if (splitText === '均分') splitText = '1/2 均分';
            if (splitText === '只有室友') splitText = `幫${paidBy === '我' ? '他' : '自己'}墊`;

            itemEl.innerHTML = `
                <div class="record-details">
                    <span class="record-date">${record.date}</span>
                    <span>${record.item} (${record.paidBy}付 / ${splitText})</span>
                </div>
                <div>
                    <span class="record-amt">$${record.amount.toFixed(0)}</span>
                    <button class="delete-btn" onclick="deleteRecord(${record.id})">刪除</button>
                </div>
            `;
            listEl.appendChild(itemEl);
        });

        // 更新結算文字
        if (netBalance > 0) {
            balanceEl.innerHTML = `目前結算：室友欠我 <span style="color: #2ecc71; font-size: 24px;">$${netBalance.toFixed(0)}</span>`;
            balanceEl.style.borderColor = '#2ecc71';
            balanceEl.style.backgroundColor = '#e8f8f5';
        } else if (netBalance < 0) {
            balanceEl.innerHTML = `目前結算：我欠室友 <span style="color: #ff7675; font-size: 24px;">$${Math.abs(netBalance).toFixed(0)}</span>`;
            balanceEl.style.borderColor = '#ff7675';
            balanceEl.style.backgroundColor = '#fff5f5';
        } else {
            balanceEl.innerHTML = `目前結算：帳目已清 👍`;
            balanceEl.style.borderColor = '#90caf9';
            balanceEl.style.backgroundColor = '#e3f2fd';
        }
    }

    // 刪除單筆紀錄
    function deleteRecord(id) {
        if (confirm('確定要刪除這筆紀錄嗎？')) {
            records = records.filter(r => r.id !== id);
            saveData();
            updateUI();
        }
    }

    // 清空所有資料
    function clearAllData() {
        if (confirm('☠️ 確定要清空所有歷史紀錄嗎？這無法還原！')) {
            records = [];
            saveData();
            updateUI();
        }
    }

    // 儲存至 localStorage
    function saveData() {
        localStorage.setItem('roommate_bills', JSON.stringify(records));
    }

    // 更新 Radio Button 的 CSS 樣式 (iOS 風格)
    function updateRadioStyling() {
        const groups = document.querySelectorAll('.radio-group');
        groups.forEach(group => {
            const options = group.querySelectorAll('.radio-option');
            options.forEach(option => {
                if (option.querySelector('input').checked) {
                    option.classList.add('selected');
                } else {
                    option.classList.remove('selected');
                }
            });
        });
    }
    
    // 第一次載入先呼叫一次樣式更新
    updateRadioStyling();
</script>
